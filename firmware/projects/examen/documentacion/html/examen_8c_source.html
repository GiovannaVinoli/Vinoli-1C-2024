110static void Sensar_pH(void *pvParameter){
111
112 while(true){
113 if(encender == true){
114 AnalogInputReadSingle(CH3, &valores_pH);
115 //convierto a valores de pH utilizando la función lineal: y = (3/14)* x
116 if((valores_pH*(3/14)) < 6){
117 GPIOOn(GPIO_23);//enciendo la bomba de pHB
118 bomba_pHB = true;
119 }
120 if((valores_pH*(3/14)) > 6.7){
121 GPIOOn(GPIO_22);//enciendo la bomba de pHA
122 bomba_pHA = true;
123 }
124 else{
125 GPIOOff(GPIO_22);
126 GPIOOff(GPIO_23);
127 bomba_pHA = false;
128 bomba_pHB = false;
129 }
130 }
131 else{
132 GPIOOff(GPIO_22);
133 GPIOOff(GPIO_23);
134 bomba_pHA = false;
135 bomba_pHB = false;
136 }
137 vTaskDelay(CONFIG_PERIOD_mS_3_SEG/portTICK_PERIOD_MS);
138 }
