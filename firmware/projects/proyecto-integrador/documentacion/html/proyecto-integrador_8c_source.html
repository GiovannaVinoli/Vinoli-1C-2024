172static void medir_tiempo(void *pParam){
173 bool timerStart = false, gpio_ant = false, enviar = false;
174 uint16_t contador=0;
175 while(true){
176 gpio = GPIORead(GPIO_3);
177 if(gpio_ant == true && gpio == false){
178 timerStart = true;
179 //printf("Comienzo a contar\n");
180 }
181 if(gpio_ant == false && gpio == true && timerStart == true){
182 timerStart = false;
183 //printf("Paro de contar\n");
184 //printf("Tiempo: %d\n", tiempo);
185 }
186 if(timerStart == true){
187 contador +=1;
188 enviar = true;
189 }
190 if(timerStart == false && enviar == true){
191 enviarDatos(contador);
192 //printf("Tiempo: %u\n", contador);
193 contador = 0;
194 enviar = false;
195 }
196 gpio_ant = gpio;
197 vTaskDelay(CONFIG_PERIOD_mS_1seg/portTICK_PERIOD_MS);
198 }
199}
